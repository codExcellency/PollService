Rest Home Page

All Polls

Method: Get Endpoint: /polls

Json Example:

[
  {
    "pollId": 3,
    "name": "Ryhmätyö-kysely",
    "description": "Anna palautetta ryhmätyöskentelyn järjestelyistä"
  }
]

Poll By ID

{
  "pollId": 3,
  "name": "Ryhmätyö-kysely",
  "description": "Anna palautetta ryhmätyöskentelyn järjestelyistä",
  "questions": [
    {
      "questionId": 4,
      "content": "Miten ryhmäjako vaikutti ryhmän toimivuuteen?",
      "questionType": "TEXT",
      "answers": [
        {
          "answerId": 1,
          "content": "Ei mitenkään"
        },
        {
          "answerId": 2,
          "content": "Oli todella mukava ratkaisu"
        }
      ],
      "options": [
        
      ],
      "tempOption": null
}
